<body>
    <p id="first"></p>

    <script>
        // Implement a RESET using the `r` key.
        // Pag nakapindot na, hindi na pwede unless i-reset.
 
        document.addEventListener('keyup', function(e) {
            console.log(e);
            const target = document.getElementById('first');
            if(e.key == 'a') {
                target.innerText = 'Nauna si A';
                target.style.backgroundColor = 'red';
                
            }
            else if(e.key == 'l') {
                target.innerText = 'Nauna si L';
                target.style.backgroundColor = 'blue';
            }
            else if(e.key == 'r') {
                target.innerText = '';
            }
 
 
        });
    </script>
</body>
